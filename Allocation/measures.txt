Measures:



Sum of Amount = SUM(Allocation[Amount])

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

Amount To Allocate = 

VAR _amt = CALCULATE(CALCULATE([Sum of Amount],FILTER(Allocation,Allocation[Account]=4990)),ALL(Allocation))



RETURN _amt

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

Amount All Other Accounts = 

VAR _amt = CALCULATE(CALCULATE([Sum of Amount],FILTER(Allocation,Allocation[Account]<>4990)),ALL(Allocation))



RETURN _amt

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

Share Of Allocation = CALCULATE( DIVIDE([Sum of Amount],[Amount All Other Accounts]),FILTER(Allocation,NOT(Allocation[Account]=4990)))

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

Allocation Amount = [Share Of Allocation]*[Amount To Allocate]
